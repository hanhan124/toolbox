
Sub TransformQPCRData()
    Dim ws As Worksheet, newWs As Worksheet
    Dim lastRow As Long
    Dim dict As Object
    Dim targetCol As Integer, sampleCol As Integer, cqCol As Integer
    Dim newRow As Long, newCol As Long
    Dim sample As Variant, target As Variant, cqValue As Variant
    
    ' 设置工作表
    Set ws = ThisWorkbook.Sheets("Sheet1") ' 请根据需要修改工作表名称
    Set newWs = ThisWorkbook.Sheets.Add
    newWs.Name = "Transformed Data"
    
    ' 找到最后一行
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    ' 找到列索引
    targetCol = WorksheetFunction.Match("Target", ws.Rows(1), 0)
    sampleCol = WorksheetFunction.Match("Sample", ws.Rows(1), 0)
    cqCol = WorksheetFunction.Match("Cq", ws.Rows(1), 0)
    
    ' 创建字典来存储数据
    Set dict = CreateObject("Scripting.Dictionary")
    
    ' 读取数据并重新整理
    Dim i As Long
    For i = 2 To lastRow
        sample = ws.Cells(i, sampleCol).Value
        target = ws.Cells(i, targetCol).Value
        cqValue = ws.Cells(i, cqCol).Value
        
        If Not dict.exists(sample) Then
            dict.Add sample, CreateObject("Scripting.Dictionary")
        End If
        
        If Not dict(sample).exists(target) Then
            Set dict(sample)(target) = New Collection  ' 改用 Collection
        End If
        
        ' 存储多个重复的 Cq 值
        dict(sample)(target).Add cqValue
    Next i
    
    ' 写入新表头
    newWs.Cells(1, 1).Value = "num"
    newWs.Cells(1, 2).Value = "group name"
    newCol = 3
    
    ' 获取所有基因名称
    Dim geneDict As Object
    Set geneDict = CreateObject("Scripting.Dictionary")
    
    For Each sample In dict.keys
        For Each target In dict(sample).keys
            If Not geneDict.exists(target) Then
                geneDict.Add target, newCol
                newWs.Cells(1, newCol).Value = target
                newCol = newCol + 1
            End If
        Next target
    Next sample
    
    ' 填充数据
    newRow = 2
    Dim maxReps As Integer, j As Integer
    
    For Each sample In dict.keys
        maxReps = 0
        ' 获取最大重复数
        For Each target In dict(sample).keys
            If dict(sample)(target).Count > maxReps Then maxReps = dict(sample)(target).Count
        Next target
        
        For j = 1 To maxReps
            If j = 1 Then
                newWs.Cells(newRow, 1).Value = newRow - 1 ' num列
                newWs.Cells(newRow, 2).Value = sample ' group name列
            End If
            
            For Each target In dict(sample).keys
                If j <= dict(sample)(target).Count Then
                    newWs.Cells(newRow, geneDict(target)).Value = dict(sample)(target).Item(j)
                End If
            Next target
            
            newRow = newRow + 1
        Next j
    Next sample
    
    ' 自动调整列宽
    newWs.Cells.EntireColumn.AutoFit
    
    ' 清理内存
    Set dict = Nothing
    Set geneDict = Nothing
    
    MsgBox "数据转换完成！", vbInformation
End Sub
